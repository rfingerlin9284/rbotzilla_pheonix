Used for Institutional Supply/Demand Liquidity Sweep Strategy. """ signals = [] for i in range(3, len(df)): highs = df['high'][i-3:i] if max(highs) - min(highs) < 0.002 * max(highs): if df['high'][i] > max(highs) and df['close'][i] < df['open'][i]: signals.append({ 'index': i, 'signal': 'sell', 'entry': df['close'][i], 'sl': df['high'][i] + 0.2, 'tp': df['close'][i] - (3 * (df['high'][i] - df['close'][i])) }) return signals def fibonacci_confluence_entry(df): """ Identifies retracements to the 50–61.8% fib zone after a breakout. Used for Break + Fib Confluence Strategy.
