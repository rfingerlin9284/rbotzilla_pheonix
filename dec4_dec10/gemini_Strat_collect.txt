        if not self.oanda_token or not self.oanda_account:

            self.results["OANDA_API"] = "⚪ SKIPPED"
            return

            "Authorization": f"Bearer {self.oanda_token}",
            "Content-Type": "application/json"
        }
        
        # 1. Auth Check (Get Accounts)
        try:
            url = f"https://api-fxpractice.oanda.com/v3/accounts/{self.oanda_account}"
            resp = requests.get(url, headers=headers, timeout=5)
            
            if resp.status_code == 200:
                self.results["OANDA_AUTH"] = "🟢 VALID"
                self.results["OANDA_API"] = "🟢 CONNECTED"
            elif resp.status_code == 401:
                self.results["OANDA_AUTH"] = "🔴 INVALID TOKEN"
                self.results["OANDA_API"] = "🔴 ACCESS DENIED"
            else:
                self.results["OANDA_AUTH"] = f"🟠 ERROR {resp.status_code}"
                self.results["OANDA_API"] = "🟠 UNSTABLE"
        except Exception as e:
            self.results["OANDA_AUTH"] = "🔴 ERROR"
            self.results["OANDA_API"] = f"🔴 {str(e)[:20]}..."

    def check_ibkr_tws(self):
        """Checks if TWS/Gateway is listening on the configured port."""
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(3)
        try:
            result = sock.connect_ex((self.ibkr_host, self.ibkr_port))
            if result == 0:
                self.results["IBKR_TWS"] = f"🟢 LISTENING (Port {self.ibkr_port})"
            else:
                self.results["IBKR_TWS"] = f"🔴 CLOSED (Port {self.ibkr_port})"
        except Exception:
            self.results["IBKR_TWS"] = "🔴 ERROR"
        finally:
            sock.close()

    def run_matrix(self):
        print("\n🔎 RUNNING CONNECTION MATRIX...\n")
        
        self.check_internet()

        self.check_ibkr_tws()

```python
import logging
import threading
import time
import queue
from ibapi.client import EClient
from ibapi.wrapper import EWrapper
from ibapi.contract import Contract
from ibapi.order import Order
from auth_manager import AuthManager

logger = logging.getLogger(__name__)

class IBKRConnector(EWrapper, EClient):
    def __init__(self, config=None):
        EClient.__init__(self, self)
        self.config = config or {}
        self.auth = AuthManager()
        self.host = self.auth.get_ibkr_host()
        self.port = self.auth.get_ibkr_port()
        self.client_id = self.auth.get_ibkr_client_id()
        self.environment = "live" if self.auth.is_live() else "paper"
        self.connected = False
        self.nextOrderId = None
        self.api_thread = None
        self.response_queue = queue.Queue()  # For handling async responses
        self.micro_trade_enabled = self.config.get('MICRO_TRADE_ENABLED', False)

    def connect(self):
        try:
            super().connect(self.host, self.port, self.client_id)
            self.api_thread = threading.Thread(target=self.run, daemon=True)
            self.api_thread.start()
            time.sleep(1)  # Wait for connection to establish
            if self.isConnected():
                self.connected = True
                logger.info(f"Connected to IBKR in {self.environment} mode.")
                self.reqIds(-1)  # Request next valid ID
            else:
                raise ConnectionError("Failed to connect to IBKR.")
        except Exception as e:
            logger.error(f"Connection failed: {e}")
            self.connected = False

    def is_connected(self):
        return self.connected and self.isConnected()

    def heartbeat(self, instruments):
        if not self.connected:
            return False, "Not connected"
        
        try:
            # Request current time as basic heartbeat
            self.reqCurrentTime()
            # Wait for response via queue (timeout 5s)
            try:
                response = self.response_queue.get(timeout=5)
                if 'time' in response:
                    # Optionally check market open for instruments
                    for symbol in instruments:
                        contract = self.create_contract(symbol)
                        self.reqMktData(self.next_req_id(), contract, '', True, False, [])
                    return True, "Market responsive"
            except queue.Empty:
                return False, "Heartbeat timeout"
        except Exception as e:
            logger.error(f"Heartbeat failed: {e}")
            return False, str(e)

    def place_order(self, order_spec):
        if not self.micro_trade_enabled and order_spec.get('timeframe', '') in ['M1', 'M5']:
            raise ValueError("Micro-trading timeframe rejected per charter")
        
        contract = self.create_contract(order_spec['symbol'], order_spec.get('secType', 'STK'), order_spec.get('exchange', 'SMART'), order_spec.get('currency', 'USD'))
        order = self.create_order(order_spec['action'], order_spec['quantity'], order_spec.get('orderType', 'MKT'))
        
        if 'limitPrice' in order_spec:
            order.lmtPrice = order_spec['limitPrice']
            order.orderType = 'LMT'
        
        # For bracket: add child orders if specified
        if 'bracket' in order_spec:
            parent_id = self.nextOrderId
            order.orderId = parent_id
            self.placeOrder(parent_id, contract, order)
            self.nextOrderId += 1
            
            # Example stop-loss child
            stop = self.create_order('SELL' if order_spec['action'] == 'BUY' else 'BUY', order_spec['quantity'], 'STP')
            stop.auxPrice = order_spec['bracket']['stopPrice']
            stop.parentId = parent_id
            stop.orderId = self.nextOrderId
            self.placeOrder(self.nextOrderId, contract, stop)
            self.nextOrderId += 1
        else:
            order.orderId = self.nextOrderId
            self.placeOrder(self.nextOrderId, contract, order)
            self.nextOrderId += 1

    def disconnect(self):
        self.disconnect()
        if self.api_thread:
            self.api_thread.join(timeout=5)
        self.connected = False
        logger.info("Disconnected from IBKR.")

    # Helper methods
    def create_contract(self, symbol, secType='STK', exchange='SMART', currency='USD'):
        contract = Contract()
        contract.symbol = symbol
        contract.secType = secType
        contract.exchange = exchange
        contract.currency = currency
        return contract

    def create_order(self, action, quantity, orderType='MKT'):
        order = Order()
        order.action = action
        order.totalQuantity = quantity
        order.orderType = orderType
        return order

    def next_req_id(self):
        # Simple incrementing reqId; can be improved
        return id(self)  # Placeholder, use proper management

    # Overrides
    def error(self, reqId, errorCode, errorString):
        logger.error(f"Error {errorCode}: {errorString} (reqId: {reqId})")
        if errorCode in [502, 504]:  # Connection issues
            self.connected = False

    def nextValidId(self, orderId):
        self.nextOrderId = orderId
        logger.info(f"Next valid order ID: {orderId}")

    def currentTime(self, time):
        self.response_queue.put({'time': time})

    def tickPrice(self, reqId, tickType, price, attrib):
        logger.info(f"Tick price: reqId={reqId}, type={tickType}, price={price}")

    def orderStatus(self, orderId, status, filled, remaining, avgFillPrice, permId, parentId, lastFillPrice, clientId, whyHeld, mktCapPrice):
        logger.info(f"Order status: ID={orderId}, status={status}, filled={filled}")
```

    def calculate_atr(high, low, close, period):

            tr2 = abs(high[i] - close[i-1])
            tr3 = abs(low[i] - close[i-1])
            tr.append(max(tr1, tr2, tr3))

        return atr

class DynamicLeverage:
    def __init__(self, max_leverage=50, volatility_window=30):
        self.max_leverage = max_leverage
        self.volatility_window = volatility_window

    def calculate_leverage(self, symbol_data: Dict, account_equity: float) -> float:
        # Calculate volatility (standard deviation of returns)
        closes = symbol_data.get('close', [])
        if len(closes) < self.volatility_window:
            return 1.0

        returns = np.diff(closes) / closes[:-1]
        volatility = np.std(returns)

        # Inverse relationship: higher volatility -> lower leverage
        # We scale the leverage between 1 and max_leverage based on volatility
        # But we want to reduce leverage when volatility is high.
        # We can use: leverage = max_leverage / (1 + volatility * scaling_factor)
        scaling_factor = 100  # Adjust this based on your risk appetite
        leverage = self.max_leverage / (1 + volatility * scaling_factor)
        leverage = max(1, min(self.max_leverage, leverage))

        return leverage

"""

```python name=test_connection_matrix.py
- Pings trading services for connectivity and authentication.
- Tests include OANDA and IBKR status for Paper mode only.

import requests
from util.auth_manager import AuthManager

BROKER_URLS = {
    "OANDA": "https://api-fxtrade.oanda.com/v3/accounts",
    "IBKR": "https://localhost:5000/v1/api/iserver/account",
}

def test_connection(broker_name):
    """Test connectivity and authentication to a broker."""
    url = BROKER_URLS.get(broker_name)
    if not url:
        return {"status": "RED", "details": "Unknown broker"}

    auth_headers = AuthManager.get_headers_for_broker(broker_name)
    try:
        response = requests.get(url, headers=auth_headers, timeout=5)
        if response.ok:
            return {"status": "GREEN", "details": "Connection successful"}
        else:
            return {"status": "RED", "details": f"HTTP {response.status_code}"}
    except Exception as e:
        return {"status": "RED", "details": str(e)}

def main():
    for broker in BROKER_URLS.keys():
        status = test_connection(broker)
        print(f"{broker}: {status['status']} ({status['details']})")

if __name__ == "__main__":
    main()
```

**auth_manager.py**
```python
import os
from dotenv import load_dotenv

class AuthManager:
    def __init__(self, env_file='.env'):
        load_dotenv(env_file)
        self.mode = os.getenv('MODE', 'PAPER').upper()
        self._validate_mode()
        self.credentials = self._load_credentials()

    def _validate_mode(self):
        if self.mode not in ['PAPER', 'LIVE']:
            raise ValueError("Invalid MODE. Must be 'PAPER' or 'LIVE'.")

    def _load_credentials(self):
        creds = {}
        if self.mode == 'LIVE':
            required_keys = ['OANDA_LIVE_TOKEN', 'OANDA_LIVE_ACCOUNT_ID', 'IBKR_HOST', 'IBKR_PORT', 'IBKR_CLIENT_ID']
        else:  # PAPER
            required_keys = ['OANDA_PAPER_TOKEN', 'OANDA_PAPER_ACCOUNT_ID', 'IBKR_PAPER_HOST', 'IBKR_PAPER_PORT', 'IBKR_PAPER_CLIENT_ID']
        
        missing = []
        for key in required_keys:
            value = os.getenv(key)
            if not value:
                missing.append(key)
            creds[key] = value
        
        if missing:
            raise ValueError(f"Missing required environment variables for {self.mode} mode: {', '.join(missing)}")
        
        return creds

    def get_oanda_token(self):
        return self.credentials['OANDA_LIVE_TOKEN' if self.mode == 'LIVE' else 'OANDA_PAPER_TOKEN']

    def get_oanda_account_id(self):
        return self.credentials['OANDA_LIVE_ACCOUNT_ID' if self.mode == 'LIVE' else 'OANDA_PAPER_ACCOUNT_ID']

    def get_ibkr_host(self):
        return self.credentials['IBKR_HOST' if self.mode == 'LIVE' else 'IBKR_PAPER_HOST']

    def get_ibkr_port(self):
        return int(self.credentials['IBKR_PORT' if self.mode == 'LIVE' else 'IBKR_PAPER_PORT'])

    def get_ibkr_client_id(self):
        return int(self.credentials['IBKR_CLIENT_ID' if self.mode == 'LIVE' else 'IBKR_PAPER_CLIENT_ID'])

    def is_live(self):
        return self.mode == 'LIVE'
```

```python
RICK_LIVE_CLEAN - Change Tracker
Singleton class for immutable change logging
AUTH CODE: 841921

import hashlib
import json
import os
from datetime import datetime
from typing import Dict, Any

class ChangeTracker:
    _instance = None
    _log_file = "change_audit.log"
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super(ChangeTracker, cls).__new__(cls)
            cls._instance._initialized = False
        return cls._instance
    
    def __init__(self):
        if not self._initialized:
            self._initialized = True
            self._ensure_log_file()
    
    def _ensure_log_file(self):
        """Create log file with header if it doesn't exist"""
        if not os.path.exists(self._log_file):
            with open(self._log_file, 'w') as f:
                f.write("# RICK_LIVE_CLEAN CHANGE AUDIT LOG\n")
                f.write(f"# Created: {datetime.utcnow().isoformat()}Z\n")
                f.write(f"# AUTH CODE: 841921\n")
                f.write("# FORMAT: timestamp|file|action|hash_before|hash_after|pin|user\n")
                f.write("#" * 80 + "\n")
    
    def _calculate_file_hash(self, file_path: str) -> str:
        """Calculate SHA256 hash of file content"""
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                content = f.read()
            return hashlib.sha256(content.encode()).hexdigest()[:16]
        except Exception as e:
            return f"ERROR:{str(e)[:20]}"
    
    def log_change(self, file_path: str, action: str, pin: str, user: str = "system") -> bool:
        """
        Log a file modification with hash verification
        
        Args:
            file_path: Path to modified file
            action: CREATE, MODIFY, DELETE
            pin: Authorization PIN (841921)
            user: User/system initiating change
            
        Returns:
            bool: Success status
        """
        if pin != "841921":
            self._log_unauthorized_attempt(file_path, action, user)
            return False
        
        timestamp = datetime.utcnow().isoformat() + 'Z'
        
        # Calculate hashes
        if action == "CREATE":
            hash_before = "NULL"
            hash_after = self._calculate_file_hash(file_path)
        elif action == "MODIFY":
            # Note: In real implementation, we'd need to capture before state
            hash_before = "CAPTURE_FAILED"  
            hash_after = self._calculate_file_hash(file_path)
        elif action == "DELETE":
            hash_before = self._calculate_file_hash(file_path)
            hash_after = "DELETED"
        else:
            hash_before = "UNKNOWN"
            hash_after = "UNKNOWN"
        
        # Log entry
        log_entry = f"{timestamp}|{file_path}|{action}|{hash_before}|{hash_after}|{pin}|{user}\n"
        
        try:
            with open(self._log_file, 'a') as f:
                f.write(log_entry)
            
            print(f"🔍 CHANGE TRACKED: {action} on {file_path} by {user}")
            return True
        except Exception as e:
            print(f"❌ CHANGE TRACKING FAILED: {e}")
            return False
    
    def _log_unauthorized_attempt(self, file_path: str, action: str, user: str):
        """Log unauthorized change attempts"""
        timestamp = datetime.utcnow().isoformat() + 'Z'
        log_entry = f"{timestamp}|{file_path}|{action}|UNAUTHORIZED|BLOCKED|INVALID_PIN|{user}\n"
```

import requests
from util.auth_manager import AuthManager

BROKER_URLS = {
    "OANDA": "https://api-fxtrade.oanda.com/v3/accounts",
    "IBKR": "https://localhost:5000/v1/api/iserver/account",
}

def test_connection(broker_name):
    """Test connectivity and authentication to a broker."""
    url = BROKER_URLS.get(broker_name)
    if not url:
        return {"status": "RED", "details": "Unknown broker"}

    auth_headers = AuthManager.get_headers_for_broker(broker_name)
    try:
        response = requests.get(url, headers=auth_headers, timeout=5)
        if response.ok:
            return {"status": "GREEN", "details": "Connection successful"}
        else:
            return {"status": "RED", "details": f"HTTP {response.status_code}"}
    except Exception as e:
        return {"status": "RED", "details": str(e)}

def main():
    for broker in BROKER_URLS.keys():
        status = test_connection(broker)
        print(f"{broker}: {status['status']} ({status['details']})")

import pandas as pd

class StrategyAggregator:
    """
    Aggregates signals; enforces timeframe filters and micro trading logic.
    """
    def __init__(self, vote_threshold=2):
        self.vote_threshold = vote_threshold

    def enforce_timeframe(self, signal):
        """Drop signals below M15 threshold."""
        timeframe = signal.get("timeframe")
        if timeframe in ["M1", "M5"]:
            print(f"NOISE TRADING REJECTED: {timeframe}")
            return False
        return True

    def aggregate_signals(self, signals):
        filtered_signals = [
            sig for sig in signals if self.enforce_timeframe(sig)

"""
execution_gate.py
- Validates trade execution (e.g., risk, leverage, modes).
"""

class ExecutionGate:
    def __init__(self, signal):
        self.signal = signal

    def evaluate(self):
        if self.signal.get("timeframe") in ["M1", "M5"]:
            return {"filter": "M15_NOISE"}
        if self.signal.get("order_size") < 1000:
            return {"filter": "MICRO_DISABLED"}
        if self.signal.get("risk", 0) > 0.05:
            return {"filter": "RISK_EXCEEDED"}
        return {"filter": "ACCEPTED"}
